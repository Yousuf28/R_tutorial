# base R, reshape function
base_wide <- data.frame(player=c('A', 'B', 'C', 'D'),
                 year1=c(12, 15, 19, 19),
                 year2=c(22, 29, 18, 12))
base_wide
  player year1 year2
1      A    12    22
2      B    15    29
3      C    19    18
4      D    19    12
# wide to long
long_df <- reshape(
  base_wide,
  direction = "long",
  varying = list(c("year1", "year2")),
  v.names = "points",
  idvar = "player",
  times = c("year1", "year2"),
  timevar = "year"
)
